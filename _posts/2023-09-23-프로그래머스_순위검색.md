```python
from itertools import combinations
from bisect import bisect_left

def solution(info, query):
    answer = []
    combi_dict = {}
    
    for line in info:
        line = line.split(" ")
        line_key = line[:-1]
        line_val = line[-1]
        
        for i in range(5):
            for combi in combinations(line_key, i):
                temp = "".join(combi)
                if temp in combi_dict:
                    combi_dict[temp].append(int(line_val))
                else:
                    combi_dict[temp] = [int(line_val)]
    for k in combi_dict:
        combi_dict[k].sort()
    
    
    for q in query:
        q = q.replace("and ", "")
        q = q.replace("- ", "")
        q = q.split(" ")
        q_key = q[:-1]
        q_val = q[-1]
        q_key = "".join(q_key)
        if q_key in combi_dict:
            vals = combi_dict[q_key]
            if vals:
                cnt = bisect_left(vals, int(q_val))
                answer.append(len(vals)-cnt)
        else:
            answer.append(0)
            
    return answer
```
